```typescript
export interface ProviderModelDef {
  readonly id: string; // e.g., 'cerebras/gpt-oss-120b'
  readonly apiId: string; // e.g., 'gpt-oss-120b' — use this for JSON models[].id
  readonly qualityTier: QualityTierType; // 'frontier' | 'high' | 'mid' | 'small'
  readonly free: boolean;
  readonly capabilities: {
    readonly toolCall: boolean; // maps to 'tools' in JSON
    readonly reasoning: boolean; // maps to 'reasoning' in JSON
    readonly vision: boolean; // maps to 'vision' in JSON
    readonly structuredOutput: boolean; // maps to 'structuredOutput' in JSON
  };
}

export interface ProviderModule {
  readonly id: string; // provider slug — use for JSON id
  readonly name: string; // display name
  readonly sdkPackage: string; // npm package name
  readonly envVar: string; // env var name
  readonly lastVerified: string; // ISO date
  readonly tier: 'free' | 'trial' | 'paid';
  readonly models: readonly ProviderModelDef[];
}
```

Capability mapping: `toolCall: true` -> `"tools"`, `reasoning: true` -> `"reasoning"`, `vision: true` -> `"vision"`, `structuredOutput: true` -> `"structuredOutput"`. Only include capabilities that are `true`.
