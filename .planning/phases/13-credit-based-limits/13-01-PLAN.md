```typescript
import type { RouterEventPayload } from '../types/events.js';

/** Credit configuration for a provider (from user config) */
export interface CreditConfig {
  balance: number; // initial $ from provider
  costRates: {
    inputPer1MTokens: number;
    outputPer1MTokens: number;
  };
  expiresAt?: string; // ISO 8601 datetime, omit for perpetual
  alertThresholds: number[]; // fractions like [0.2, 0.05] = fire at 20% and 5% remaining
  expiryWarningDays: number; // default 7
}

/** Credit status for a provider (from getStatus()) */
export interface CreditStatus {
  provider: string;
  balance: number; // configured total balance
  consumed: number; // total consumed in dollars
  remaining: number; // balance - consumed
  percentUsed: number; // (consumed / balance) * 100
  expiresAt?: string; // ISO string if configured
  expired: boolean; // true if past expiresAt
  exhausted: boolean; // true if remaining <= 0 AND confirmed by 402
}

/** credit:low event payload */
export interface CreditLowEvent extends RouterEventPayload {
  provider: string;
  threshold: number; // the threshold that was crossed (e.g. 0.2)
  remaining: number; // remaining dollars
  balance: number; // configured balance
  percentRemaining: number; // remaining/balance * 100
}

/** credit:exhausted event payload */
export interface CreditExhaustedEvent extends RouterEventPayload {
  provider: string;
  balance: number;
  consumed: number;
  reason: 'depleted' | 'expired';
}

/** credit:expiring event payload */
export interface CreditExpiringEvent extends RouterEventPayload {
  provider: string;
  expiresAt: string;
  daysRemaining: number;
}
```

   ```typescript
   export function createCreditLimit(options: {
     balance: number;

     expiresAt?: string;
     alertThresholds?: number[];
     expiryWarningDays?: number;
   ```
