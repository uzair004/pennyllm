**1. Create `src/fallback/AffinityCache.ts`**:

Simple TTL cache for short-term affinity (avoids repeated fallback resolution during burst traffic):

```typescript
import type { AffinityEntry } from './types.js';

export class AffinityCache {
  private cache: Map<string, AffinityEntry> = new Map();
  private readonly ttlMs: number;

  constructor(ttlMs: number = 60_000) {
    this.ttlMs = ttlMs;
  }

  get(key: string): AffinityEntry | null {
    const entry = this.cache.get(key);
    if (!entry) return null;
    if (Date.now() - entry.timestamp > this.ttlMs) {
      this.cache.delete(key);
      return null;
    }
    return entry;
  }

  set(key: string, provider: string, modelId: string): void {
    this.cache.set(key, { provider, modelId, timestamp: Date.now() });
  }

  clear(): void {
    this.cache.clear();
  }
}
```

**Helper: `isFallbackTrigger(error: unknown): boolean`**

Returns true if the error should trigger cross-provider fallback:

- `error instanceof QuotaExhaustedError` (all keys quota-exhausted)
- `error instanceof RateLimitError` (all keys rate-limited)
- `error instanceof ProviderError` and its `errorType` is `'server'` (server errors trigger cross-provider fallback per user decision)

Import `QuotaExhaustedError`, `RateLimitError`, `ProviderError` from `../errors/index.js`.

**Helper: `getProviderBehavior(provider: string): FallbackBehavior`**

1. Check per-provider override: `deps.config.providers[provider]?.fallback?.behavior`
2. Fall back to global: `deps.config.fallback.behavior`
3. Default: `'auto'`

**Helper: `buildProviderAttempt(provider: string, modelId: string, error: unknown): ProviderAttempt`**

Build a `ProviderAttempt` record from an error. Determine `reason` from error type:

- `QuotaExhaustedError` -> `'quota_exhausted'`
- `RateLimitError` -> `'rate_limited'`
- `ProviderError` with server type -> `'server_error'`
- Otherwise -> `'quota_exhausted'` as fallback reason

Extract `earliestRecovery` from error metadata if available. Use conditional property inclusion for optional fields.

**doGenerate implementation:**

```
1. Determine required capabilities:
   - Call FallbackResolver.inferCapabilities(params, deps.reasoning) to get caps from request
   - Get original model metadata from catalog for quality tier

2. Check per-provider fallback behavior:
   - If getProviderBehavior(primaryProvider) === 'hard-stop', just call primaryRetryProxy.doGenerate(params) and let any error propagate (no fallback)

3. Check if fallback is globally enabled:
   - If !deps.config.fallback.enabled, call primaryRetryProxy.doGenerate(params) directly

4. Build affinity key: `${primaryProvider}:${JSON.stringify(requiredCaps)}`
5. Check affinity cache for cached fallback

6. Try primary provider first:
   try {
     const result = await deps.primaryRetryProxy.doGenerate(params);
     return result;
   } catch (error) {
     if (!isFallbackTrigger(error)) throw error;
     // Primary failed with fallback-triggerable error
     triedProviders.push(buildProviderAttempt(primaryProvider, primaryModelId, error));
   }

7. Emit provider:exhausted event (with exhaustionType field):
   - Determine exhaustionType from error type
   - Emit via safeEmit

8. If affinity cache hit AND cached provider not in excluded list:
   - Try the cached provider first (create retry proxy for it)
   - On success: update refs, emit fallback:triggered, return result
   - On failure: add to triedProviders, continue

9. Resolve fallback candidates via resolver:
   const candidates = await deps.fallbackResolver.resolve(
     deps.primaryModelId,
     requiredCaps,
     {
       configuredProviders: Object.keys(deps.config.providers).filter(p => deps.config.providers[p]?.enabled !== false),
       excludeProviders: triedProviders.map(a => a.provider),
       budgetRemaining: await deps.budgetTracker.getRemainingBudget(),
       estimatedTokens: deps.estimatedTokens,
       originalQualityTier: originalModelMeta?.qualityTier,
     }
   );

10. Try each candidate up to maxDepth - 1 (primary was attempt 1):
    for (const candidate of candidates.slice(0, deps.config.fallback.maxDepth - 1)) {
      // Budget gate: if model is not free and budget is exceeded, skip
      if (!candidate.isFree) {
        const budgetExceeded = await deps.budgetTracker.isExceeded();
        if (budgetExceeded) {
          triedProviders.push({ provider: candidate.provider, modelId: candidate.modelId, reason: 'budget_exceeded' });
          continue;
        }
      }

      // Select a key for this fallback provider
      try {
        const fallbackSelection = await deps.keySelector.selectKey(
          candidate.provider, candidate.modelName, { requestId: deps.requestId }
        );
        const providerConfig = deps.config.providers[candidate.provider];
        if (!providerConfig) continue;
        const keyConfig = providerConfig.keys[fallbackSelection.keyIndex];
        if (keyConfig === undefined) continue;
        const apiKey = typeof keyConfig === 'string' ? keyConfig : keyConfig.key;

        // Create a fresh base model for fallback provider
        const fallbackModel = createProviderInstance(deps.registry, candidate.provider, candidate.modelName, apiKey);

        // Create retry proxy for the fallback provider (each gets its own)
        const fallbackKeyIndexRef = { current: fallbackSelection.keyIndex };
        const fallbackRetryProxy = createRetryProxy({
          provider: candidate.provider,
          modelName: candidate.modelName,
          modelId: candidate.modelId,
          initialModel: fallbackModel,
          initialKeyIndex: fallbackSelection.keyIndex,
          initialKey: apiKey,
          config: deps.config,
          registry: deps.registry,
          cooldownManager: deps.cooldownManager,
          disabledKeys: deps.disabledKeys,
          emitter: deps.emitter,
          requestId: deps.requestId,
          keyIndexRef: fallbackKeyIndexRef,
          keySelector: deps.keySelector,
        });

        const result = await fallbackRetryProxy.doGenerate(params);

        // SUCCESS! Update all refs for middleware to record against correct provider
        deps.providerRef.current = candidate.provider;
        deps.modelIdRef.current = candidate.modelId;
        deps.keyIndexRef.current = fallbackKeyIndexRef.current;

        // Update affinity cache
        deps.affinityCache.set(affinityKey, candidate.provider, candidate.modelId);

        // Emit fallback:triggered event
        safeEmit(deps.emitter, RouterEvent.FALLBACK_TRIGGERED, {
          fromProvider: deps.primaryProvider,
          toProvider: candidate.provider,
          fromModel: deps.primaryModelId,
          toModel: candidate.modelId,
          reason: triedProviders[0]?.reason ?? 'unknown',
          requestId: deps.requestId,
          timestamp: Date.now(),
        });

        // Add fallback metadata to response via providerMetadata
        // This augments without breaking the AI SDK response contract
        if (result.providerMetadata) {
          result.providerMetadata['llm-router'] = {
            fallbackUsed: true,
            originalModel: deps.primaryModelId,
            actualModel: candidate.modelId,
          };
        }

        // Record cost for budget tracking (fire-and-forget)
        const usage = result.usage;
        deps.budgetTracker.recordCost(
          candidate.provider,
          candidate.modelId,
          {
            promptTokens: Number(usage?.inputTokens?.total) || 0,
            completionTokens: Number(usage?.outputTokens?.total) || 0,
          },
          candidate.pricing,
        ).catch(err => debug('Budget recording failed: %s', err instanceof Error ? err.message : String(err)));

        return result;
      } catch (error) {
        triedProviders.push(buildProviderAttempt(candidate.provider, candidate.modelId, error));
        continue;
      }
    }

11. All providers exhausted:
    throw new AllProvidersExhaustedError(deps.primaryModelId, triedProviders);
```
