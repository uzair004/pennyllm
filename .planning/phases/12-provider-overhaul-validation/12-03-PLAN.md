```typescript
export const configSchema = z
  .object({
    version: z.literal('1.0').default('1.0'),
    providers: z
      .record(z.string(), providerConfigSchema)
      .refine((obj) => Object.keys(obj).length > 0, {
        message: 'At least one provider is required',
      }),
    strategy: z
      .enum([Strategy.PRIORITY, Strategy.ROUND_ROBIN, Strategy.LEAST_USED] as const)
      .default(Strategy.PRIORITY),
    budget: budgetConfigSchema.default({
      monthlyLimit: 0,
      alertThresholds: [0.8, 0.95],
    }),
    // NEW: explicit model priority chain (overrides auto-generation when present)
    models: z.array(z.string()).optional(),
    estimation: estimationSchema,
    cooldown: cooldownSchema,
    warningThreshold: z.number().min(0).max(1).optional(),
    applyRegistryDefaults: z.boolean().default(false),
    dryRun: z.boolean().default(false),
    debug: z.boolean().default(false),
    // REMOVED: fallback: fallbackConfigSchema
  })
  .strict()
  .refine(
    (config) => {
      // Validate: trial tier providers must have credits field
      for (const [name, prov] of Object.entries(config.providers)) {
        if (prov.tier === 'trial' && prov.credits === undefined) {
          return false; // Will use the message below
        }
      }
      return true;
    },
    {
      message: "Providers with tier 'trial' must specify a 'credits' value.",
    },
  );
```

Remove `fallbackConfigSchema` and `providerFallbackOverrideSchema` exports entirely (delete those schemas from the file).
