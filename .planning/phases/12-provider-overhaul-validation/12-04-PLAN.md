 * Built at createRouter() time, immutable for the session.
 */
export interface ChainEntry {
  /** Provider ID e.g. 'cerebras' */
  provider: string;
  /** Full model ID e.g. 'cerebras/llama-4-maverick' */
  modelId: string;
  /** API model ID e.g. 'llama-4-maverick' (what SDK expects) */
  apiModelId: string;
  /** Quality tier for filtering */
  qualityTier: QualityTierType;
  /** Whether this is a free-tier model */
  free: boolean;
  /** Model capabilities for per-request filtering */
  capabilities: ProviderModelDef['capabilities'];
  /** Whether this model has been 404'd this session (mutable) */
  stale: boolean;
}

/**
 * Record of a single chain attempt (for error reporting).
 */
export interface ChainAttempt {
  provider: string;
  modelId: string;
  chainPosition: number;
  errorType: string;
  statusCode?: number;
  cooldownMs?: number;
  cooldownClass?: CooldownClass;
  message: string;
}

/**
 * Result from successful chain execution.

export interface ChainResult {

```typescript
import debugFactory from 'debug';
import type { RouterConfig } from '../types/config.js';
import type { ChainEntry } from './types.js';
import { getProviderModule } from '../providers/registry.js';
import type { ProviderModelDef } from '../providers/types.js';
import type { QualityTierType } from '../constants/index.js';

const debug = debugFactory('pennyllm:chain');

/** Quality tier sort order (frontier first) */
const TIER_ORDER: Record<QualityTierType, number> = {
  frontier: 0,
  high: 1,
  mid: 2,
  small: 3,
};
```

**Implement `buildChain(config: RouterConfig): ChainEntry[]`:**

Two modes based on whether `config.models` is present:

**Mode 1: Explicit chain** (config.models is defined and non-empty)

For each model ID in `config.models`:

1. Parse `provider/model` format. Extract provider name.
2. Verify provider is configured in `config.providers`. If not, log warning: `"Model ${id} references unconfigured provider '${provider}', skipping"` and skip.
3. Look up the ProviderModule via `getProviderModule(provider)`.
4. Find the model in the module's curated models array by matching `id`.
5. If found: create ChainEntry with metadata from curated model (qualityTier, free, capabilities, apiModelId).
6. If NOT found in curated models: allow anyway (unknown model per user decision). Use apiModelId = model part after slash. Set `qualityTier: 'mid'`, `free: true` (assume free), capabilities all false. Log warning: `"Model ${id} not in known registry for provider '${provider}', will attempt anyway"`.
7. If provider has a per-provider `models` array (allowlist): check model is in it. If not, throw ConfigError: `"Model '${model}' is not in the allowlist for provider '${provider}'. Allowed: [${allowed.join(', ')}]"`.
8. Set `stale: false`.

**Mode 2: Auto-generated chain** (config.models is undefined or empty)

1. Collect configured providers, sorted by `priority` (lower first, then alphabetically for ties).
2. For each provider, get its ProviderModule. If no module found, log warning and skip.
3. For each provider, get the models to use:
   - If provider has per-provider `models` array: use those (in order), looking up metadata from curated registry.
   - Otherwise: use all curated models from the ProviderModule.
4. Filter models: if `config.budget.monthlyLimit === 0`, include only free models. If budget > 0, include free AND paid.
5. Sort each provider's models by quality tier (frontier first via TIER_ORDER).
6. **Interleave** across providers: round-robin by provider order, then by model position within each provider. Algorithm:
   ```
   maxModels = max(providerModels.length for all providers)
   for modelIndex 0..maxModels-1:
     for each provider in priority order:
       if provider has a model at modelIndex:
         add to chain
   ```
7. Log the full chain at startup.

**Implement `logChain(chain: ChainEntry[]): void`:**

Always log (not just debug mode). Use `console.info`:
